// $in => Matches any of the values specified in an array.
db.series.find({
   "Ano de lançamento": { $in: [2019,2015] }
}).project({"Serie":1, "Ano de lançamento":1})

// $nin => Matches none of the values specified in an array.
db.series.find({
    "Ano de lançamento": {$nin:[2015,2019]}
}).project({"Serie":1, "Ano de lançamento":1})
